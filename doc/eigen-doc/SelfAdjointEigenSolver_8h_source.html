388 template<typename MatrixType>
389 SelfAdjointEigenSolver<MatrixType>& SelfAdjointEigenSolver<MatrixType>
390 ::compute(const MatrixType& matrix, int options)
391 {
392  check_template_parameters();
393 
394  using std::abs;
395  eigen_assert(matrix.cols() == matrix.rows());
396  eigen_assert((options&~(EigVecMask|GenEigMask))==0
397  && (options&EigVecMask)!=EigVecMask
398  && "invalid option parameter");
399  bool computeEigenvectors = (options&ComputeEigenvectors)==ComputeEigenvectors;
400  Index n = matrix.cols();
401  m_eivalues.resize(n,1);
402 
403  if(n==1)
404  {
405  m_eivalues.coeffRef(0,0) = numext::real(matrix.coeff(0,0));
406  if(computeEigenvectors)
407  m_eivec.setOnes(n,n);
408  m_info = Success;
409  m_isInitialized = true;
410  m_eigenvectorsOk = computeEigenvectors;
411  return *this;
412  }
413 
414  // declare some aliases
415  RealVectorType& diag = m_eivalues;
416  MatrixType& mat = m_eivec;
417 
418  // map the matrix coefficients to [-1:1] to avoid over- and underflow.
419  mat = matrix.template triangularView<Lower>();
420  RealScalar scale = mat.cwiseAbs().maxCoeff();
421  if(scale==RealScalar(0)) scale = RealScalar(1);
422  mat.template triangularView<Lower>() /= scale;
423  m_subdiag.resize(n-1);
424  internal::tridiagonalization_inplace(mat, diag, m_subdiag, computeEigenvectors);
425 
426  Index end = n-1;
427  Index start = 0;
428  Index iter = 0; // total number of iterations
429 
430  while (end>0)
431  {
432  for (Index i = start; i<end; ++i)
433  if (internal::isMuchSmallerThan(abs(m_subdiag[i]),(abs(diag[i])+abs(diag[i+1]))))
434  m_subdiag[i] = 0;
435 
436  // find the largest unreduced block
437  while (end>0 && m_subdiag[end-1]==0)
438  {
439  end--;
440  }
441  if (end<=0)
442  break;
443 
444  // if we spent too many iterations, we give up
445  iter++;
446  if(iter > m_maxIterations * n) break;
447 
448  start = end - 1;
449  while (start>0 && m_subdiag[start-1]!=0)
450  start--;
451 
452  internal::tridiagonal_qr_step<MatrixType::Flags&RowMajorBit ? RowMajor : ColMajor>(diag.data(), m_subdiag.data(), start, end, computeEigenvectors ? m_eivec.data() : (Scalar*)0, n);
453  }
454 
455  if (iter <= m_maxIterations * n)
456  m_info = Success;
457  else
458  m_info = NoConvergence;
459 
460  // Sort eigenvalues and corresponding vectors.
461  // TODO make the sort optional ?
462  // TODO use a better sort algorithm !!
463  if (m_info == Success)
464  {
465  for (Index i = 0; i < n-1; ++i)
466  {
467  Index k;
468  m_eivalues.segment(i,n-i).minCoeff(&k);
469  if (k > 0)
470  {
471  std::swap(m_eivalues[i], m_eivalues[k+i]);
472  if(computeEigenvectors)
473  m_eivec.col(i).swap(m_eivec.col(k+i));
474  }
475  }
476  }
477 
478  // scale back the eigen values
479  m_eivalues *= scale;
480 
481  m_isInitialized = true;
482  m_eigenvectorsOk = computeEigenvectors;
483  return *this;
484 }

501  static inline void computeRoots(const MatrixType& m, VectorType& roots)
502  {
503  using std::sqrt;
504  using std::atan2;
505  using std::cos;
506  using std::sin;
507  const Scalar s_inv3 = Scalar(1.0)/Scalar(3.0);
508  const Scalar s_sqrt3 = sqrt(Scalar(3.0));
509 
510  // The characteristic equation is x^3 - c2*x^2 + c1*x - c0 = 0. The
511  // eigenvalues are the roots to this equation, all guaranteed to be
512  // real-valued, because the matrix is symmetric.
513  Scalar c0 = m(0,0)*m(1,1)*m(2,2) + Scalar(2)*m(1,0)*m(2,0)*m(2,1) - m(0,0)*m(2,1)*m(2,1) - m(1,1)*m(2,0)*m(2,0) - m(2,2)*m(1,0)*m(1,0);
514  Scalar c1 = m(0,0)*m(1,1) - m(1,0)*m(1,0) + m(0,0)*m(2,2) - m(2,0)*m(2,0) + m(1,1)*m(2,2) - m(2,1)*m(2,1);
515  Scalar c2 = m(0,0) + m(1,1) + m(2,2);
516 
517  // Construct the parameters used in classifying the roots of the equation
518  // and in solving the equation for the roots in closed form.
519  Scalar c2_over_3 = c2*s_inv3;
520  Scalar a_over_3 = (c1 - c2*c2_over_3)*s_inv3;
521  if (a_over_3 > Scalar(0))
522  a_over_3 = Scalar(0);
523 
524  Scalar half_b = Scalar(0.5)*(c0 + c2_over_3*(Scalar(2)*c2_over_3*c2_over_3 - c1));
525 
526  Scalar q = half_b*half_b + a_over_3*a_over_3*a_over_3;
527  if (q > Scalar(0))
528  q = Scalar(0);
529 
530  // Compute the eigenvalues by solving for the roots of the polynomial.
531  Scalar rho = sqrt(-a_over_3);
532  Scalar theta = atan2(sqrt(-q),half_b)*s_inv3;
533  Scalar cos_theta = cos(theta);
534  Scalar sin_theta = sin(theta);
535  roots(0) = c2_over_3 + Scalar(2)*rho*cos_theta;
536  roots(1) = c2_over_3 - rho*(cos_theta + s_sqrt3*sin_theta);
537  roots(2) = c2_over_3 - rho*(cos_theta - s_sqrt3*sin_theta);
538 
539  // Sort in increasing order.
540  if (roots(0) >= roots(1))
541  std::swap(roots(0),roots(1));
542  if (roots(1) >= roots(2))
543  {
544  std::swap(roots(1),roots(2));
545  if (roots(0) >= roots(1))
546  std::swap(roots(0),roots(1));
547  }
548  }

550  static inline void run(SolverType& solver, const MatrixType& mat, int options)
551  {
552  using std::sqrt;
553  eigen_assert(mat.cols() == 3 && mat.cols() == mat.rows());
554  eigen_assert((options&~(EigVecMask|GenEigMask))==0
555  && (options&EigVecMask)!=EigVecMask
556  && "invalid option parameter");
557  bool computeEigenvectors = (options&ComputeEigenvectors)==ComputeEigenvectors;
558 
559  MatrixType& eivecs = solver.m_eivec;
560  VectorType& eivals = solver.m_eivalues;
561 
562  // map the matrix coefficients to [-1:1] to avoid over- and underflow.
563  Scalar scale = mat.cwiseAbs().maxCoeff();
564  MatrixType scaledMat = mat / scale;
565 
566  // compute the eigenvalues
567  computeRoots(scaledMat,eivals);
568 
569  // compute the eigen vectors
570  if(computeEigenvectors)
571  {
572  Scalar safeNorm2 = Eigen::NumTraits<Scalar>::epsilon();
573  if((eivals(2)-eivals(0))<=Eigen::NumTraits<Scalar>::epsilon())
574  {
575  eivecs.setIdentity();
576  }
577  else
578  {
579  scaledMat = scaledMat.template selfadjointView<Lower>();
580  MatrixType tmp;
581  tmp = scaledMat;
582 
583  Scalar d0 = eivals(2) - eivals(1);
584  Scalar d1 = eivals(1) - eivals(0);
585  int k = d0 > d1 ? 2 : 0;
586  d0 = d0 > d1 ? d0 : d1;
587 
588  tmp.diagonal().array () -= eivals(k);
589  VectorType cross;
590  Scalar n;
591  n = (cross = tmp.row(0).cross(tmp.row(1))).squaredNorm();
592 
593  if(n>safeNorm2)
594  {
595  eivecs.col(k) = cross / sqrt(n);
596  }
597  else
598  {
599  n = (cross = tmp.row(0).cross(tmp.row(2))).squaredNorm();
600 
601  if(n>safeNorm2)
602  {
603  eivecs.col(k) = cross / sqrt(n);
604  }
605  else
606  {
607  n = (cross = tmp.row(1).cross(tmp.row(2))).squaredNorm();
608 
609  if(n>safeNorm2)
610  {
611  eivecs.col(k) = cross / sqrt(n);
612  }
613  else
614  {
615  // the input matrix and/or the eigenvaues probably contains some inf/NaN,
616  // => exit
617  // scale back to the original size.
618  eivals *= scale;
619 
620  solver.m_info = NumericalIssue;
621  solver.m_isInitialized = true;
622  solver.m_eigenvectorsOk = computeEigenvectors;
623  return;
624  }
625  }
626  }
627 
628  tmp = scaledMat;
629  tmp.diagonal().array() -= eivals(1);
630 
631  if(d0<=Eigen::NumTraits<Scalar>::epsilon())
632  {
633  eivecs.col(1) = eivecs.col(k).unitOrthogonal();
634  }
635  else
636  {
637  n = (cross = eivecs.col(k).cross(tmp.row(0))).squaredNorm();
638  if(n>safeNorm2)
639  {
640  eivecs.col(1) = cross / sqrt(n);
641  }
642  else
643  {
644  n = (cross = eivecs.col(k).cross(tmp.row(1))).squaredNorm();
645  if(n>safeNorm2)
646  eivecs.col(1) = cross / sqrt(n);
647  else
648  {
649  n = (cross = eivecs.col(k).cross(tmp.row(2))).squaredNorm();
650  if(n>safeNorm2)
651  eivecs.col(1) = cross / sqrt(n);
652  else
653  {
654  // we should never reach this point,
655  // if so the last two eigenvalues are likely to be very close to each other
656  eivecs.col(1) = eivecs.col(k).unitOrthogonal();
657  }
658  }
659  }
660 
661  // make sure that eivecs[1] is orthogonal to eivecs[2]
662  // FIXME: this step should not be needed
663  Scalar d = eivecs.col(1).dot(eivecs.col(k));
664  eivecs.col(1) = (eivecs.col(1) - d * eivecs.col(k)).normalized();
665  }
666 
667  eivecs.col(k==2 ? 0 : 2) = eivecs.col(k).cross(eivecs.col(1)).normalized();
668  }
669  }
670  // Rescale back to the original size.
671  eivals *= scale;
672 
673  solver.m_info = Success;
674  solver.m_isInitialized = true;
675  solver.m_eigenvectorsOk = computeEigenvectors;
676  }

679 // 2x2 direct eigenvalues decomposition, code from Hauke Heibel
680 template<typename SolverType> struct direct_selfadjoint_eigenvalues<SolverType,2,false>
681 {
682  typedef typename SolverType::MatrixType MatrixType;
683  typedef typename SolverType::RealVectorType VectorType;
684  typedef typename SolverType::Scalar Scalar;
685 
686  static inline void computeRoots(const MatrixType& m, VectorType& roots)
687  {
688  using std::sqrt;
689  const Scalar t0 = Scalar(0.5) * sqrt( numext::abs2(m(0,0)-m(1,1)) + Scalar(4)*m(1,0)*m(1,0));
690  const Scalar t1 = Scalar(0.5) * (m(0,0) + m(1,1));
691  roots(0) = t1 - t0;
692  roots(1) = t1 + t0;
693  }
694 
695  static inline void run(SolverType& solver, const MatrixType& mat, int options)
696  {
697  using std::sqrt;
698  using std::abs;
699 
700  eigen_assert(mat.cols() == 2 && mat.cols() == mat.rows());
701  eigen_assert((options&~(EigVecMask|GenEigMask))==0
702  && (options&EigVecMask)!=EigVecMask
703  && "invalid option parameter");
704  bool computeEigenvectors = (options&ComputeEigenvectors)==ComputeEigenvectors;
705 
706  MatrixType& eivecs = solver.m_eivec;
707  VectorType& eivals = solver.m_eivalues;
708 
709  // map the matrix coefficients to [-1:1] to avoid over- and underflow.
710  Scalar scale = mat.cwiseAbs().maxCoeff();
711  scale = (std::max)(scale,Scalar(1));
712  MatrixType scaledMat = mat / scale;
713 
714  // Compute the eigenvalues
715  computeRoots(scaledMat,eivals);
716 
717  // compute the eigen vectors
718  if(computeEigenvectors)
719  {
720  if((eivals(1)-eivals(0))<=abs(eivals(1))*Eigen::NumTraits<Scalar>::epsilon())
721  {
722  eivecs.setIdentity();
723  }
724  else
725  {
726  scaledMat.diagonal().array () -= eivals(1);
727  Scalar a2 = numext::abs2(scaledMat(0,0));
728  Scalar c2 = numext::abs2(scaledMat(1,1));
729  Scalar b2 = numext::abs2(scaledMat(1,0));
730  if(a2>c2)
731  {
732  eivecs.col(1) << -scaledMat(1,0), scaledMat(0,0);
733  eivecs.col(1) /= sqrt(a2+b2);
734  }
735  else
736  {
737  eivecs.col(1) << -scaledMat(1,1), scaledMat(1,0);
738  eivecs.col(1) /= sqrt(c2+b2);
739  }
740 
741  eivecs.col(0) << eivecs.col(1).unitOrthogonal();
742  }
743  }
744 
745  // Rescale back to the original size.
746  eivals *= scale;
747 
748  solver.m_info = Success;
749  solver.m_isInitialized = true;
750  solver.m_eigenvectorsOk = computeEigenvectors;
751  }
752 };

765 template<int StorageOrder,typename RealScalar, typename Scalar, typename Index>
766 static void tridiagonal_qr_step(RealScalar* diag, RealScalar* subdiag, Index start, Index end, Scalar* matrixQ, Index n)
767 {
768  using std::abs;
769  RealScalar td = (diag[end-1] - diag[end])*RealScalar(0.5);
770  RealScalar e = subdiag[end-1];
771  // Note that thanks to scaling, e^2 or td^2 cannot overflow, however they can still
772  // underflow thus leading to inf/NaN values when using the following commented code:
773 // RealScalar e2 = numext::abs2(subdiag[end-1]);
774 // RealScalar mu = diag[end] - e2 / (td + (td>0 ? 1 : -1) * sqrt(td*td + e2));
775  // This explain the following, somewhat more complicated, version:
776  RealScalar mu = diag[end];
777  if(td==0)
778  mu -= abs(e);
779  else
780  {
781  RealScalar e2 = numext::abs2(subdiag[end-1]);
782  RealScalar h = numext::hypot(td,e);
783  if(e2==0) mu -= (e / (td + (td>0 ? 1 : -1))) * (e / h);
784  else mu -= e2 / (td + (td>0 ? h : -h));
785  }
786 
787  RealScalar x = diag[start] - mu;
788  RealScalar z = subdiag[start];
789  for (Index k = start; k < end; ++k)
790  {
791  JacobiRotation<RealScalar> rot;
792  rot.makeGivens(x, z);
793 
794  // do T = G' T G
795  RealScalar sdk = rot.s() * diag[k] + rot.c() * subdiag[k];
796  RealScalar dkp1 = rot.s() * subdiag[k] + rot.c() * diag[k+1];
797 
798  diag[k] = rot.c() * (rot.c() * diag[k] - rot.s() * subdiag[k]) - rot.s() * (rot.c() * subdiag[k] - rot.s() * diag[k+1]);
799  diag[k+1] = rot.s() * sdk + rot.c() * dkp1;
800  subdiag[k] = rot.c() * sdk - rot.s() * dkp1;
801 
802 
803  if (k > start)
804  subdiag[k - 1] = rot.c() * subdiag[k-1] - rot.s() * z;
805 
806  x = subdiag[k];
807 
808  if (k < end - 1)
809  {
810  z = -rot.s() * subdiag[k+1];
811  subdiag[k + 1] = rot.c() * subdiag[k+1];
812  }
813 
814  // apply the givens rotation to the unit matrix Q = Q * G
815  if (matrixQ)
816  {
817  // FIXME if StorageOrder == RowMajor this operation is not very efficient
818  Map<Matrix<Scalar,Dynamic,Dynamic,StorageOrder> > q(matrixQ,n,n);
819  q.applyOnTheRight(k,k+1,rot);
820  }
821  }
822 }
