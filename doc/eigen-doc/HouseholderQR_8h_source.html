224 template<typename MatrixQR, typename HCoeffs>
225 void householder_qr_inplace_unblocked(MatrixQR& mat, HCoeffs& hCoeffs, typename MatrixQR::Scalar* tempData = 0)
226 {
227  typedef typename MatrixQR::Index Index;
228  typedef typename MatrixQR::Scalar Scalar;
229  typedef typename MatrixQR::RealScalar RealScalar;
230  Index rows = mat.rows();
231  Index cols = mat.cols();
232  Index size = (std::min)(rows,cols);
233 
234  eigen_assert(hCoeffs.size() == size);
235 
236  typedef Matrix<Scalar,MatrixQR::ColsAtCompileTime,1> TempType;
237  TempType tempVector;
238  if(tempData==0)
239  {
240  tempVector.resize(cols);
241  tempData = tempVector.data();
242  }
243 
244  for(Index k = 0; k < size; ++k)
245  {
246  Index remainingRows = rows - k;
247  Index remainingCols = cols - k - 1;
248 
249  RealScalar beta;
250  mat.col(k).tail(remainingRows).makeHouseholderInPlace(hCoeffs.coeffRef(k), beta);
251  mat.coeffRef(k,k) = beta;
252 
253  // apply H to remaining part of m_qr from the left
254  mat.bottomRightCorner(remainingRows, remainingCols)
255  .applyHouseholderOnTheLeft(mat.col(k).tail(remainingRows-1), hCoeffs.coeffRef(k), tempData+k+1);
256  }
257 }

260 template<typename MatrixQR, typename HCoeffs>
261 void householder_qr_inplace_blocked(MatrixQR& mat, HCoeffs& hCoeffs,
262  typename MatrixQR::Index maxBlockSize=32,
263  typename MatrixQR::Scalar* tempData = 0)
264 {
265  typedef typename MatrixQR::Index Index;
266  typedef typename MatrixQR::Scalar Scalar;
267  typedef Block<MatrixQR,Dynamic,Dynamic> BlockType;
268 
269  Index rows = mat.rows();
270  Index cols = mat.cols();
271  Index size = (std::min)(rows, cols);
272 
273  typedef Matrix<Scalar,Dynamic,1,ColMajor,MatrixQR::MaxColsAtCompileTime,1> TempType;
274  TempType tempVector;
275  if(tempData==0)
276  {
277  tempVector.resize(cols);
278  tempData = tempVector.data();
279  }
280 
281  Index blockSize = (std::min)(maxBlockSize,size);
282 
283  Index k = 0;
284  for (k = 0; k < size; k += blockSize)
285  {
286  Index bs = (std::min)(size-k,blockSize); // actual size of the block
287  Index tcols = cols - k - bs; // trailing columns
288  Index brows = rows-k; // rows of the block
289 
290  // partition the matrix:
291  // A00 | A01 | A02
292  // mat = A10 | A11 | A12
293  // A20 | A21 | A22
294  // and performs the qr dec of [A11^T A12^T]^T
295  // and update [A21^T A22^T]^T using level 3 operations.
296  // Finally, the algorithm continue on A22
297 
298  BlockType A11_21 = mat.block(k,k,brows,bs);
299  Block<HCoeffs,Dynamic,1> hCoeffsSegment = hCoeffs.segment(k,bs);
300 
301  householder_qr_inplace_unblocked(A11_21, hCoeffsSegment, tempData);
302 
303  if(tcols)
304  {
305  BlockType A21_22 = mat.block(k,k+bs,brows,tcols);
306  apply_block_householder_on_the_left(A21_22,A11_21,hCoeffsSegment.adjoint());
307  }
308  }
309 }

311 template<typename _MatrixType, typename Rhs>
312 struct solve_retval<HouseholderQR<_MatrixType>, Rhs>
313  : solve_retval_base<HouseholderQR<_MatrixType>, Rhs>
314 {
315  EIGEN_MAKE_SOLVE_HELPERS(HouseholderQR<_MatrixType>,Rhs)
316 
317  template<typename Dest> void evalTo(Dest& dst) const
318  {
319  const Index rows = dec().rows(), cols = dec().cols();
320  const Index rank = (std::min)(rows, cols);
321  eigen_assert(rhs().rows() == rows);
322 
323  typename Rhs::PlainObject c(rhs());
324 
325  // Note that the matrix Q = H_0^* H_1^*... so its inverse is Q^* = (H_0 H_1 ...)^T
326  c.applyOnTheLeft(householderSequence(
327  dec().matrixQR().leftCols(rank),
328  dec().hCoeffs().head(rank)).transpose()
329  );
330 
331  dec().matrixQR()
332  .topLeftCorner(rank, rank)
333  .template triangularView<Upper>()
334  .solveInPlace(c.topRows(rank));
335 
336  dst.topRows(rank) = c.topRows(rank);
337  dst.bottomRows(cols-rank).setZero();
338  }
339 };
